511 static inline std::string get_back_trace()
512 {
513  std::string bt_str;
514 #if defined(MIPP_ENABLE_BACKTRACE) && (defined(__GNUC__) || defined(__clang__) || defined(__llvm__)) && (defined(__linux__) || defined(__linux) || defined(__APPLE__)) && !defined(__ANDROID__)
515  const int bt_max_depth = 32;
516  void *bt_array[bt_max_depth];
517 
518  size_t size = backtrace(bt_array, bt_max_depth);
519  char** bt_symbs = backtrace_symbols(bt_array, size);
520 
521  bt_str += "\nBacktrace:";
522  for (size_t i = 0; i < size; i++)
523  bt_str += "\n" + std::string(bt_symbs[i]);
524  free(bt_symbs);
525 #endif
526 
527  return bt_str;
528 }
